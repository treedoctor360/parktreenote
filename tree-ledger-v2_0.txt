// =============================================
//  IndexedDB 初期化
// =============================================
var DB_NAME = 'TreeLedger2', STORE = 'records', DB_VER = 1;
var db = null;

function initDB(cb) {
  var req = indexedDB.open(DB_NAME, DB_VER);
  req.onupgradeneeded = function(e) {
    var d = e.target.result;
    if (!d.objectStoreNames.contains(STORE))
      d.createObjectStore(STORE, { keyPath: 'id', autoIncrement: true });
  };
  req.onsuccess = function(e) { db = e.target.result; cb(); };
  req.onerror   = function()  { alert('IndexedDB初期化失敗'); };
}
function dbGetAll(cb) {
  db.transaction(STORE,'readonly').objectStore(STORE).getAll().onsuccess = function(e){ cb(e.target.result); };
}
function dbAdd(rec, cb) {
  var req = db.transaction(STORE,'readwrite').objectStore(STORE).add(rec);
  req.onsuccess = function(){ cb && cb(req.result); };
}
function dbPut(rec, cb) {
  db.transaction(STORE,'readwrite').objectStore(STORE).put(rec).onsuccess = function(){ cb && cb(); };
}
function dbDelete(id, cb) {
  db.transaction(STORE,'readwrite').objectStore(STORE).delete(id).onsuccess = function(){ cb && cb(); };
}

// =============================================
//  テキスト解析
//  書式: # 公園名 → セクション開始
//        樹種　[幹周ランク]　[数量]　[メモ]
// =============================================
var parsedRows = []; // 解析結果を一時保持

function parseText() {
  var raw = document.getElementById('raw-text').value.trim();
  if (!raw) { alert('テキストを入力してください'); return; }

  var lines = raw.split(/\r?\n/);
  var rows = [];
  var currentPark = '';

  lines.forEach(function(line) {
    line = line.trim();
    if (!line) return;

    // 公園名行: # で始まる
    if (/^#/.test(line)) {
      currentPark = line.replace(/^#+\s*/, '').trim();
      return;
    }
    // 「入り口から」などナビゲーション注釈行はスキップ
    if (/^入り口|^奥から|^右側|^左側|^北|^南|^東|^西/.test(line)) return;

    // --- 樹種・幹周ランク・数量・メモを抽出 ---
    var parsed = parseLine(line, currentPark);
    if (parsed) rows.push(parsed);
  });

  if (!rows.length) { alert('解析できる行がありませんでした'); return; }

  parsedRows = rows;
  renderPreview(rows);
}

function parseLine(line, park) {
  // 数量パターン: "N本" または "N本=Xm2"
  var qtyMatch  = line.match(/(\d+)\s*本/);
  var qty       = qtyMatch ? parseInt(qtyMatch[1]) : 1;

  // 面積パターン: "= X m2" または "Xm2" または "X✖️Y" など
  var areaMatch = line.match(/[=＝]?\s*([\d.]+)\s*m2/i)
               || line.match(/([\d.]+)\s*[mｍ]2/i);
  var area = areaMatch ? areaMatch[1] + 'm2' : '';

  // 幹周ランクパターン: c数字-数字 or c数字以上/以下 or h数字以上 など
  var girthMatch = line.match(/[cCｃＣ]\s*[\d.]+\s*[-‐–]\s*[\d.]+/i)   // c60-90
                || line.match(/[cCｃＣ]\s*[\d.]+\s*(以上|以下|[+＋])/i) // c90以上
                || line.match(/[cCｃＣ]\s*[\d.]+/i)                      // c90
                || line.match(/[hHｈＨ]\s*[\d.]+\s*(以上|以下)/i);       // h150以上

  // 幹周ランクを正規化（c60-90 などを抜き出す）
  var girth = '';
  if (girthMatch) {
    girth = girthMatch[0].replace(/\s/g,'').replace(/ｃ/g,'c').replace(/ＣＣ/g,'c');
  }

  // 樹種: 行頭から最初のスペース・タブ・数字区切りまで
  // ただし幹周ランク・数量・計算式の前まで
  var speciesRaw = line
    .replace(/\s*[cCｃＣhHｈＨ][\d.\s\-–以上以下+＋×✖️xX*]+/gi, ' ')
    .replace(/\s*\d+\s*本[^\s]*/gi, ' ')
    .replace(/\s*[=＝]\s*[\d.]+\s*m2/gi, ' ')
    .replace(/\s*[\d.]+\s*m2/gi, ' ')
    .replace(/\s*w[\d.]+/gi, ' ')
    .replace(/\s*d[\d.]+/gi, ' ')
    .replace(/\s*[\d]{2,}\s*[✖️×xX\*]\s*[\d.]+/g, ' ')
    .replace(/\s+/g, ' ')
    .trim();

  // 先頭の単語（スペース区切り最初の要素）を樹種とする
  var species = speciesRaw.split(/[\s　]+/)[0].trim();
  if (!species) return null;

  // メモ: 面積、w/d寸法、その他残りの情報をまとめる
  var memoParts = [];
  var wMatch = line.match(/w\s*[\d.]+/gi);
  var dMatch = line.match(/d\s*[\d.]+/gi);
  if (wMatch) memoParts.push(wMatch.join(' '));
  if (dMatch) memoParts.push(dMatch.join(' '));
  if (area)   memoParts.push(area);

  // 株立などの表記
  if (/株立/.test(line))    memoParts.push('株立');
  if (/伐採/.test(line))    memoParts.push('伐採');
  if (/勝手生え/.test(line)) memoParts.push('勝手生え');
  if (/以上/.test(line) && !/[cCｃＣhHｈＨ]\d/.test(line.split('以上')[0].slice(-3))) {
    // 高さ表記(h150以上など)は girth に入るので重複排除
  }

  return {
    park:    park,
    species: species,
    girth:   girth,
    qty:     qty,
    memo:    memoParts.join(' / ')
  };
}

// プレビューテーブルを描画
function renderPreview(rows) {
  var tbody = document.getElementById('preview-tbody');
  var html = '';
  var lastPark = '';

  rows.forEach(function(r, i) {
    // 公園名が変わったら仕切り行を挿入
    if (r.park !== lastPark) {
      html += '<tr class="park-header"><td colspan="6">📍 ' + escHtml(r.park) + '</td></tr>';
      lastPark = r.park;
    }
    html += '<tr id="pr-' + i + '">'
      + '<td><input value="' + escAttr(r.park)    + '" onchange="updatePRow(' + i + ',\'park\',this.value)"></td>'
      + '<td><input value="' + escAttr(r.species)  + '" onchange="updatePRow(' + i + ',\'species\',this.value)"></td>'
      + '<td><input value="' + escAttr(r.girth)    + '" onchange="updatePRow(' + i + ',\'girth\',this.value)"></td>'
      + '<td><input type="number" min="1" value="' + (r.qty||1) + '" onchange="updatePRow(' + i + ',\'qty\',this.value)"></td>'
      + '<td><input value="' + escAttr(r.memo)     + '" onchange="updatePRow(' + i + ',\'memo\',this.value)"></td>'
      + '<td><button class="del-row-btn" onclick="delPreviewRow(' + i + ')">✕</button></td>'
      + '</tr>';
  });

  tbody.innerHTML = html;
  document.getElementById('preview-count').textContent = '解析結果: ' + rows.length + ' 件（修正してから登録してください）';
  document.getElementById('preview-section').style.display = 'block';
  document.getElementById('preview-section').scrollIntoView({ behavior: 'smooth', block: 'start' });
}

function updatePRow(i, key, val) {
  if (parsedRows[i]) parsedRows[i][key] = val;
}
function delPreviewRow(i) {
  parsedRows.splice(i, 1);
  renderPreview(parsedRows);
}
function addPreviewRow() {
  parsedRows.push({ park: '', species: '', girth: '', qty: 1, memo: '' });
  renderPreview(parsedRows);
  // 追加した行の公園名にフォーカス
  var rows = document.querySelectorAll('#preview-tbody tr:not(.park-header)');
  if (rows.length) {
    var last = rows[rows.length - 1];
    var inp = last.querySelector('input');
    if (inp) inp.focus();
  }
}
function clearPreview() {
  parsedRows = [];
  document.getElementById('preview-section').style.display = 'none';
  document.getElementById('raw-text').value = '';
}

// プレビューを確定してDBに保存
function commitPreview() {
  var valid = parsedRows.filter(function(r) { return r.park && r.species; });
  if (!valid.length) { alert('公園名と樹種が入力されている行が必要です'); return; }

  var done = 0;
  var now = new Date().toISOString();
  valid.forEach(function(r) {
    dbAdd({
      park:    r.park.trim(),
      species: r.species.trim(),
      girth:   r.girth  || '',
      qty:     parseInt(r.qty) || 1,
      memo:    r.memo   || '',
      photo:   null,
      date:    now
    }, function() {
      done++;
      if (done === valid.length) {
        clearPreview();
        loadAndRender();
        toast('✅ ' + done + ' 件を登録しました');
      }
    });
  });
}

var currentPhotoData = null;

function compressPhoto(file, maxPx, quality, cb) {
  var reader = new FileReader();
  reader.onload = function(e) {
    var img = new Image();
    img.onload = function() {
      var w = img.width, h = img.height;
      if (w > maxPx || h > maxPx) {
        if (w > h) { h = Math.round(h * maxPx / w); w = maxPx; }
        else       { w = Math.round(w * maxPx / h); h = maxPx; }
      }
      var c = document.createElement('canvas');
      c.width = w; c.height = h;
      c.getContext('2d').drawImage(img, 0, 0, w, h);
      cb(c.toDataURL('image/jpeg', quality));
    };
    img.src = e.target.result;
  };
  reader.readAsDataURL(file);
}

function handlePhoto(event) {
  var file = event.target.files[0];
  if (!file) return;
  compressPhoto(file, 1200, 0.75, function(dataUrl) {
    currentPhotoData = dataUrl;
    var prev = document.getElementById('photo-preview');
    prev.src = dataUrl; prev.style.display = 'block';
    document.getElementById('photo-placeholder').style.display = 'none';
    document.getElementById('photo-clear-btn').style.display = 'block';
  });
}
function clearPhoto(e) {
  if (e) e.stopPropagation();
  currentPhotoData = null;
  document.getElementById('photo-preview').style.display = 'none';
  document.getElementById('photo-preview').src = '';
  document.getElementById('photo-placeholder').style.display = 'block';
  document.getElementById('photo-clear-btn').style.display = 'none';
  document.getElementById('e-photo').value = '';
}

var currentFilter = 'all';
var allRecords = [];

function loadAndRender() {
  dbGetAll(function(records) {
    allRecords = records;
    updateTabs(records);
    updateStats(records);
    renderCards(records);
    updateDataLists(records);
  });
}

function filterByPark(park, el) {
  currentFilter = park;
  document.querySelectorAll('.tab').forEach(function(t) { t.classList.remove('active'); });
  el.classList.add('active');
  renderCards(allRecords);
}

function updateTabs(records) {
  var parks = [];
  records.forEach(function(r) { if (r.park && parks.indexOf(r.park) < 0) parks.push(r.park); });
  parks.sort();
  var el = document.getElementById('park-tabs');
  el.innerHTML = '<div class="tab' + (currentFilter==='all'?' active':'') + '" data-park="all" onclick="filterByPark(\'all\',this)">すべて</div>';
  parks.forEach(function(p) {
    var a = currentFilter === p ? ' active' : '';
    el.innerHTML += '<div class="tab' + a + '" onclick="filterByPark(\'' + escJs(p) + '\',this)">' + escHtml(p) + '</div>';
  });
}

function updateStats(records) {
  var parks = [], total = 0;
  records.forEach(function(r) {
    if (r.park && parks.indexOf(r.park) < 0) parks.push(r.park);
    total += parseInt(r.qty) || 0;
  });
  document.getElementById('stat-parks').textContent   = parks.length;
  document.getElementById('stat-trees').textContent   = total;
  document.getElementById('stat-records').textContent = records.length;
}

function renderCards(records) {
  var filtered = currentFilter === 'all' ? records : records.filter(function(r) { return r.park === currentFilter; });
  var grid  = document.getElementById('card-grid');
  var empty = document.getElementById('empty-msg');

  if (!filtered.length) { grid.innerHTML = ''; empty.style.display = 'block'; return; }
  empty.style.display = 'none';

  grid.innerHTML = filtered.map(function(r) {
    var photoArea = r.photo
      ? '<img class="card-photo-thumb" src="' + r.photo + '" alt="写真">'
      : '<div class="card-photo-empty" onclick=\'openEditModal(' + JSON.stringify(r) + ')\' title="タップして写真を追加">📷</div>';

    var metas = [];
    if (r.girth) metas.push(r.girth);
    if (r.qty)   metas.push(r.qty + '本');

    return '<div class="tree-card">'
      + photoArea
      + '<div class="card-body">'
      + '<div class="card-park">📍 ' + escHtml(r.park) + '</div>'
      + '<div class="card-species">' + escHtml(r.species) + '</div>'
      + '<div class="card-meta">' + metas.map(function(m) { return '<span>' + escHtml(m) + '</span>'; }).join('') + '</div>'
      + (r.memo ? '<div class="card-memo">' + escHtml(r.memo) + '</div>' : '')
      + '<div class="card-actions">'
      + '<button class="btn-primary btn-sm" onclick=\'openEditModal(' + JSON.stringify(r) + ')\'>✏️ 編集</button>'
      + '</div></div></div>';
  }).join('');
}

function updateDataLists(records) {
  var parks = [], species = [];
  records.forEach(function(r) {
    if (r.park    && parks.indexOf(r.park)       < 0) parks.push(r.park);
    if (r.species && species.indexOf(r.species)  < 0) species.push(r.species);
  });
  ['park-list','park-list2'].forEach(function(id) {
    document.getElementById(id).innerHTML = parks.map(function(p){ return '<option value="' + escAttr(p) + '">'; }).join('');
  });
  ['species-list','species-list2'].forEach(function(id) {
    document.getElementById(id).innerHTML = species.map(function(s){ return '<option value="' + escAttr(s) + '">'; }).join('');
  });
}

function exportCSV() {
  dbGetAll(function(records) {
    if (!records.length) { alert('データがありません'); return; }
    var rows = [['ID','公園名','樹種','幹周ランク','数量','メモ','登録日']];
    records.forEach(function(r) {
      rows.push([r.id, r.park, r.species, r.girth||'', r.qty||'', (r.memo||'').replace(/\n/g,' '), r.date ? r.date.slice(0,10) : '']);
    });
    var csv = '\uFEFF' + rows.map(function(row) {
      return row.map(function(c) {
        var s = String(c);
        return (s.includes(',') || s.includes('"') || s.includes('\n')) ? '"' + s.replace(/"/g,'""') + '"' : s;
      }).join(',');
    }).join('\r\n');
    dlFile(csv, 'text/csv', '樹木台帳_' + dStr() + '.csv');
    toast('📊 CSVをダウンロード');
  });
}

function exportJSON() {
  dbGetAll(function(records) {
    if (!records.length) { alert('データがありません'); return; }
    dlFile(JSON.stringify({ version: 2, exported: new Date().toISOString(), records }, null, 2),
      'application/json', '樹木台帳バックアップ_' + dStr() + '.json');
    toast('💾 JSONをダウンロード');
  });
}

function importJSON(event) {
  var file = event.target.files[0];
  if (!file) return;
  var reader = new FileReader();
  reader.onload = function(e) {
    try {
      var data = JSON.parse(e.target.result);
      var list = data.records || (Array.isArray(data) ? data : null);
      if (!list) { alert('形式が正しくありません'); return; }
      if (!confirm(list.length + ' 件をインポートします。よろしいですか？')) return;
      var done = 0;
      var tx = db.transaction(STORE, 'readwrite');
      var store = tx.objectStore(STORE);
      list.forEach(function(r) {
        var rec = Object.assign({}, r); delete rec.id;
        store.add(rec).onsuccess = function() {
          done++;
          if (done === list.length) { loadAndRender(); toast('📥 ' + done + ' 件インポート'); }
        };
      });
    } catch(err) { alert('読み込み失敗: ' + err.message); }
  };
  reader.readAsText(file);
  event.target.value = '';
}

function dlFile(content, mime, filename) {
  var a = document.createElement('a');
  a.href = URL.createObjectURL(new Blob([content], { type: mime }));
  a.download = filename; a.click();
  URL.revokeObjectURL(a.href);
}
function dStr() {
  var d = new Date();
  return d.getFullYear() + ('0'+(d.getMonth()+1)).slice(-2) + ('0'+d.getDate()).slice(-2);
}
function escHtml(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;').replace(/'/g,'&#039;');
}
function escAttr(s) { return escHtml(s); }
function escJs(s)   { return String(s).replace(/\\/g,'\\\\').replace(/'/g,"\\'"); }
function toast(msg) {
  var el = document.getElementById('toast');
  el.textContent = msg; el.classList.add('show');
  setTimeout(function(){ el.classList.remove('show'); }, 2500);
}
